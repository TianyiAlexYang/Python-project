The estimation of $\mathbf{w}$ is:

$\mathbf{w} = (X^TX)^-1X^TY$

def find_parameters(x, y, k):
    '''
    x:array    y:array   k:int
    A function designed to find the paramters of scalar to scalar regression, k is the largest monomial degree 
    '''
    x = x.reshape(np.shape(x)[0],)                 #adjustment for data x
    vander = np.vander(x, k+1, increasing = "T")   # construct matrix for x
    p1 = np.linalg.inv(vander.T@vander)            #calculate X.T X)^-1
    y = y.reshape(np.shape(y)[0],)                 #adjustment for data y
    p2 = vander.T@y                                #calculate (X.T Y)
    w = p1@p2                                      #parameter w is estimated by (X.T X)^-1(X.T Y)
    return w

The training error is the error between desired output and a predicted output calculated by using the same data that trained the model:

$\large E_{training} = \frac{1}{N_{training}} \sum_{p=1}^{N_{training}}E^P$

and $E^P$ is:

$\large E^P = \frac{1}{2} \sum_{o =1}^{N_o}(d_o^p - y_o^p)^2$

since $N_o = 1$, therefore $E^P$ is:

$\large E_{training} = \frac{1}{N_{training}} \sum_{p=1}^{N_{training}} \frac{1}{2}(d^p - y^p)^2$

The test error is the error between desired output and predicted output calculated by the model trained with the training dataset: $\large E_{test} = \frac{1}{N_{test}} \sum_{p=1}^{N_{test}} \frac{1}{2}(d^p - y^p)^2$

def train_error(x, y):
    '''
    x:array   y:array
    give a set of inputs and outputs, calculate training error of each degree of model
    '''
    train_error = np.zeros(11)                              #set up the train error container
    n = np.shape(y)[0]                                      #get the size of data
    for k in range(1,11):
        w = find_parameters(x, y, k)                        #calculate model of each degree
        x = x.reshape(np.shape(x)[0],)                      #adjust input for calculation
        vander = np.vander(x, k+1, increasing = "T")        #calculate corresponding degree of x matrix
        yp = vander@w                                       #calculate predicted output
        yd = y.reshape(np.shape(y)[0],)                     #adjust actual output for calculatioin
        train_error[k] = (1/n)*sum((1/2)*(yd - yp)**2)      #calculate the training error
    return np.log(train_error[1:])                              #return the log-training error
    


def test_error(xtrain, ytrain, xt, yt):
    '''
    xtrain, ytrain, xt, yt: array
    calculate the test error of each degree of model
    '''
    test_error = np.zeros(11)                               #set up the test error container            
    n = np.shape(yt)[0]                                     #get the size of test data
    for k in range(1,11):
        w = find_parameters(xtrain, ytrain, k)              #calculate model based on training set
        xt = xt.reshape(np.shape(xt)[0],)                   #adjust test input for calculation
        vander = np.vander(xt, k+1, increasing = "T")       #calculate corresponding degree of x matrix
        yp = vander@w                                       #calculate predicted output
        yd = yt.reshape(np.shape(yt)[0],)                   #adjust test output for calculation
        test_error[k] = (1/n)*sum((1/2)*(yd - yp)**2)       #calculate test error
    return np.log(test_error[1:])                               #return the log-test error                       



def visualize_error(error1, error2, k, name1, name2):
    '''
    error1, error2: array     
    k, name1, name2:string
    plot the error by complexity of the model
    '''
    plt.plot(range(1,11), error1, 'bo-', label = name1)
    plt.plot(range(1,11), error2, 'ro-', label = name2)
    plt.xlabel("model degree")
    plt.ylabel("log error")
    plt.title(k)
    plt.legend()
    plt.show()

A way of calculating cross validation with a dataset is:

$\large CV = \frac{1}{N} \sum_{i=1}^{N}(\frac{y_i - \hat{y}_i}{1 - h_i})^2$

where $\hat{y}$ is equal to $Hy$, and $h_i$ is the ith diagonal entry of matrix $H = X(X^TX)^{-1}X^T$

def find_H(x, k):
    '''
    x:array  k:int
    A function for finding H in order to calculate the cross validation score 
    '''
    x = x.reshape(np.shape(x)[0],)
    vander = np.vander(x, k+1, increasing = "T")
    inv = np.linalg.inv(vander.T@vander)
    p1 = np.matmul(vander, inv)
    h = p1@vander.T
    return h




def cv(y, h):
    '''
    y,h: array
    Calculate the cross validation value based on the formula of CV
    '''
    cv = 0
    yp = h@y
    n = np.shape(y)[0]
    for i in range(n):
        cv += ((y[i] - yp[i])/(1 - h[i,i]))**2
    return (1/n)*cv
       
def compare_cv(y, x):
    '''
    y,x:array
    returns a list of cv for degree 1 to 10 model
    '''
    cross_valid = np.zeros((11, 2))
    for i in range(1,11):
        h = find_H(x, i)
        cross_valid[i,0] = i
        cross_valid[i,1] = cv(y, h)
    return cross_valid
